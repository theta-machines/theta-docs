\begin{multicols}{2}
    \textbf{A[3:0]} -- \textit{Analog Pins} -- The pins used for analog inputs

    \textbf{ADC} -- \textit{Analog to Digital Converter} -- An integrated circuit that converts an analog signal into a digital signal
    
    \textbf{CBSEL[1:0]} -- \textit{Cold Boot Select Pins} -- Pins that can used to select which bitstream is loaded when the FPGA is in Cold Boot mode

    \textbf{CS} -- \textit{Chip Select} -- A pin used on a SPI bus for selecting the slave device

    \textbf{FPGA} -- \textit{Field Programmable Gate Array} -- An integrated circuit that can be programmed to implement digital logic

    \textbf{GBIN[7:0]} -- \textit{Global Input Pins} -- Dedicated input pins that can be used for clock signals, reset signals, and other high-fanout nets

    \textbf{GPIO} -- \textit{General Purpose Input/Output} -- Pins that can be configured as a digital input or output

    \textbf{HAT} -- \textit{Hardware Attached on Top} -- An add-on board for the Raspberry Pi to expand its functionality

    \textbf{HDL} -- \textit{Hardware Description Language} -- A computer language used to describe digital logic circuits (e.g., SystemVerilog, VHDL)
    
    \textbf{I$^2$C} -- \textit{Inter-Integrated Circuit} -- A two-wire synchronous serial communication bus

    \textbf{I/O} -- \textit{Input/Output} -- A device's input and output pins

    \textbf{LED} -- \textit{Light Emitting Diode} -- A semiconductor device that emits light

    \textbf{MISO} -- \textit{Master In Slave Out} -- One of the SPI bus data lines

    \textbf{MOSI} -- \textit{Master Out Slave In} -- One of the SPI bus data lines

    \textbf{MUX} -- \textit{Multiplexer} -- A device that selects which of many input signals to use as the output signal

    \textbf{PCB} -- \textit{Printed Circuit Board} -- A board with conductive traces and pads connecting components together

    \vfill\null

    \textbf{PLL} -- \textit{Phase-Locked Loop} -- A circuit capable of producing a higher-frequency clock signal from a lower-frequency clock signal

    \textbf{RAM} -- \textit{Random Access Memory} -- Memory that can be quickly accessed at any address

    \textbf{RTMC} -- \textit{Real-Time Motion Control} -- An architecture defining simple, decoupled interfaces between realtime and near-realtime devices with the goal of driving CNC-based machines

    \textbf{S[1:0]} -- \textit{Warm Boot Select Ports} -- These internal ports are used to select which bitstream to restart into when the FPGA is in Warm Boot Mode

    \textbf{SCK} -- \textit{Serial Clock} -- This data sheet uses SCK to define the serial clock on a SPI bus

    \textbf{SCL} -- \textit{Serial Clock} -- This data sheet uses SCL to define the serial clock on an I$^2$C bus

    \textbf{SDA} -- \textit{Serial Data} -- The bidirectional data line used on an I$^2$C bus

    \textbf{SPI} -- \textit{Serial Peripheral Interface} -- A four-wire (minimum) synchronous serial communication bus

    \textbf{TCP/IP} -- \textit{Internet Protocol Suite} -- A set of communication protocols such as TCP, UDP, ICMP, IPv4, IGMP, AND PPPoE
    
    \textbf{UART} -- \textit{Universal Asynchronous Receiver/Transmitter} -- A two-wire asynchronous serial communication bus

    \textbf{USB} -- \textit{Universal Serial Bus} -- A standard that specifies the interfaces and protocols for connecting and powering devices

\end{multicols}
